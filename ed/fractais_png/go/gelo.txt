package main

import (
	"fmt"
	"math/rand"
)

func ri(inf, sup int) float64 {
	return float64(rand.Intn(sup-inf+1) + inf)
}

func gelo(pen *Pen, raio float64, nivel int) {
	if nivel == 0 {
		return
	}
	
	for i := 0; i < 5; i++ {
		pen.Right(72)

		pen.Walk(raio)

		gelo(pen, raio*0.35, nivel-1)

		pen.Walk(-raio)
	}
}

func main() {
	pen := NewPen(1200, 1200)
	pen.SetPosition(600, 600)
	gelo(pen, 320, 6)
	pen.SavePNG("tree.png")
	fmt.Println("PNG file created successfully.")
}
